grammar music_play;


public <music_play> =
	(<musicCommand> | <musicIntent>) <musicObject> [please];


// action rules

<musicCommand> =
	[can you | could you] [please] (play [me | us]| put on);


<musicIntent> = 
	[i | we] (want | would like | d like) to (listen [to]| hear); 

	// based on the input format may be changed to: [i | we | i'd | we'd] (want | [would] like) to (listen [to]| hear);


<musicObject> = [<album> by] <artist> | 
	<album> |
	<song> | 
	[some] <genre> [music];


// music object rules
